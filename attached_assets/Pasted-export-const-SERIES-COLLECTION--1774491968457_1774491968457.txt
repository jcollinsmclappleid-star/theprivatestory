/**
 * Get complete prompt for any episode
 */
export function getEpisodePrompt(seriesId, episodeNumber) {
  const series = SERIES_COLLECTION.find(s => s.id === seriesId);
  if (!series) return null;
  const episode = series.episodes.find(e => e.episode === episodeNumber);
  if (!episode) return null;

  return {
    system: `${MASTER_SYSTEM_PROMPT}

SERIES CONTEXT:
Title: ${series.title}
Tagline: ${series.tagline}
World: ${series.world || 'As described in episode prompt'}
Series Arc: ${series.series_arc || 'As described in episode prompt'}
Male Lead: ${JSON.stringify(series.male_lead || series.female_lead)}
Female Lead: ${JSON.stringify(series.female_lead || series.male_lead)}`,

    user: `${episode.prompt}

PREVIOUS EPISODE CLIFFHANGER TO OPEN FROM:
${episodeNumber > 1 
  ? series.episodes.find(e => e.episode === episodeNumber - 1)?.cliffhanger || 'See series arc'
  : 'This is the first episode — open cold on the world.'}

SERIES POSITION: Episode ${episodeNumber} of ${series.episodes.length}
EXPLICIT LEVEL TARGET: ${episode.explicit_level}/5
WORD COUNT TARGET: ${episode.word_count_target || '1,900-2,200'}`,

    metadata: {
      series: series.title,
      episode: episodeNumber,
      title: episode.title,
      tagline: episode.tagline,
      explicit_level: episode.explicit_level,
      arc_function: episode.arc_function
    }
  };
}

/**
 * Get series overview for UI display
 */
export function getSeriesOverview(seriesId) {
  const series = SERIES_COLLECTION.find(s => s.id === seriesId);
  if (!series) return null;
  return {
    id: series.id,
    title: series.title,
    tagline: series.tagline,
    description: series.description,
    episode_count: series.episodes.length,
    episodes: series.episodes.map(e => ({
      number: e.episode,
      title: e.title,
      tagline: e.tagline,
      explicit_level: e.explicit_level,
      arc_function: e.arc_function
    }))
  };
}
