// ─── UTILITY FUNCTIONS ────────────────────────────────────────────────────────

/**
 * Build a complete LLM prompt for a given subtheme
 * Replaces [USER_INPUT] with custom text if provided
 */
export function buildPrompt(categoryId, subthemeId, userInput = null) {
  const category = STORY_CATEGORIES.find(c => c.id === categoryId);
  if (!category) return null;

  const subtheme = category.subthemes.find(s => s.id === subthemeId);
  if (!subtheme) return null;

  let storyPrompt = subtheme.prompt;

  if (subtheme.is_custom && userInput) {
    storyPrompt = storyPrompt.replace('[USER_INPUT]', userInput);
  }

  return {
    system: category.system_prompt,
    user: storyPrompt,
    metadata: {
      category: category.name,
      subtheme: subtheme.name,
      mood: category.mood,
      explicit_level: category.explicit_level,
      duration_range: subtheme.duration_range,
      intensity: subtheme.intensity,
      tags: subtheme.tags
    }
  };
}

/**
 * Get all custom subthemes across categories
 */
export function getCustomSubthemes() {
  return STORY_CATEGORIES.flatMap(cat =>
    cat.subthemes
      .filter(s => s.is_custom)
      .map(s => ({ ...s, category_id: cat.id, category_name: cat.name }))
  );
}

/**
 * Filter categories by explicit level
 * Useful for free vs premium tier gating
 */
export function getCategoriesByExplicitLevel(level) {
  const levels = {
    romantic: ['romantic'],
    suggestive: ['romantic', 'suggestive', 'romantic_to_suggestive'],
    sensual: ['romantic', 'suggestive', 'romantic_to_suggestive', 'sensual', 'suggestive_to_sensual', 'sensual_to_explicit'],
    explicit: 'all'
  };
  if (level === 'explicit') return STORY_CATEGORIES;
  return STORY_CATEGORIES.filter(c => levels[level]?.includes(c.explicit_level));
}

/**
 * Get subthemes by mood tag
 */
export function getSubthemesByTag(tag) {
  return STORY_CATEGORIES.flatMap(cat =>
    cat.subthemes
      .filter(s => s.tags?.includes(tag))
      .map(s => ({ ...s, category_id: cat.id, category_name: cat.name }))
  );
}

/**
 * Get a random story prompt — useful for discovery features
 */
export function getRandomPrompt(explicitLevel = 'sensual') {
  const available = getCategoriesByExplicitLevel(explicitLevel);
  const randomCat = available[Math.floor(Math.random() * available.length)];
  const nonCustom = randomCat.subthemes.filter(s => !s.is_custom);
  const randomSub = nonCustom[Math.floor(Math.random() * nonCustom.length)];
  return buildPrompt(randomCat.id, randomSub.id);
}
